训练流程如算法\ref{alg:binary_search}所示。
\begin{algorithm}[ht]
\caption{二分查找算法}
\label{alg:binary_search}
\KwIn{升序数组 $A[1..n]$，目标值 $x$}
\KwOut{目标下标（不存在则返回 $-1$）}
$l \leftarrow 1,\ r \leftarrow n$\;
\While{$l \le r$}{
    $mid \leftarrow \lfloor (l+r)/2 \rfloor$\;
    \uIf{$A[mid] = x$}{
        \Return $mid$\;
    }
    \uElseIf{$A[mid] < x$}{
        $l \leftarrow mid + 1$\;
    }
    \Else{
        $r \leftarrow mid - 1$\;
    }
}
\Return $-1$\;
\end{algorithm}
